public Map<TopicPartition, List<ConsumerRecord<K, V>>> fetchedRecords() {
    if (this.subscriptions.partitionAssignmentNeeded()) {
        return Collections.emptyMap();
    } else {
        Map<TopicPartition, List<ConsumerRecord<K, V>>> drained = new HashMap<>();
        int recordsRemaining = maxPollRecords;
        Iterator<CompletedFetch> completedFetchesIterator = completedFetches.iterator();

        while (recordsRemaining > 0) {
            if (nextInLineRecords == null || nextInLineRecords.isEmpty()) {
                if (!completedFetchesIterator.hasNext())
                    break;

                CompletedFetch completion = completedFetchesIterator.next();
                completedFetchesIterator.remove();
                // 解析数据。
                nextInLineRecords = parseFetchedData(completion);
            } else {
                recordsRemaining -= append(drained, nextInLineRecords, recordsRemaining);
            }
        }

        return drained;
    }
}

public static class RecordsIterator extends AbstractIterator<LogEntry> {
    private final ByteBuffer buffer;
    private final DataInputStream stream;
    private final CompressionType type;
    private final boolean shallow;
    private RecordsIterator innerIter;

    // The variables for inner iterator
    private final ArrayDeque<LogEntry> logEntries;
    private final long absoluteBaseOffset;

    public RecordsIterator(ByteBuffer buffer, boolean shallow) {
        this.type = CompressionType.NONE;
        this.buffer = buffer;
        this.shallow = shallow;
        this.stream = new DataInputStream(new ByteBufferInputStream(buffer));
        this.logEntries = null;
        this.absoluteBaseOffset = -1;
    }

重点关注shallow字段和innerIter字段。这里做初始化时，shallow为false，它表示该迭代器是深层迭代器（若为true则表示浅层迭代器）。由于这里并不知道消息有没有经过压缩，很可能需要迭代处理，因此这里创建的是一个深层迭代器，也就是外层迭代。如果需要内层的深层迭代，则需要在后面某一步给innerIter字段做初始化。

protected LogEntry makeNext() {
    // 1.判断内层迭代结束没有。
    if (innerDone()) {
        try {
            // 2.获取下一个LogEntry对象。
            LogEntry entry = getNextEntry();
            // 3.没有下一个LogEntry对象时标记state状态为DONE。
            if (entry == null)
                return allDone();

            // 4.将entry中的offset值转变为absolute offset值。
            if (absoluteBaseOffset >= 0) {
                long absoluteOffset = absoluteBaseOffset + entry.offset();
                entry = new LogEntry(absoluteOffset, entry.record());
            }

            // 5.判断是否要进一步迭代。不需要的话直接返回entry对象，否则返回内层迭代器的next方法返回值。
            CompressionType compression = entry.record().compressionType();
            if (compression == CompressionType.NONE || shallow) {
                return entry;
            } else {
                innerIter = new RecordsIterator(entry);
                return innerIter.next();
            }
        } catch (EOFException e) {
            return allDone();
        } catch (IOException e) {
            throw new KafkaException(e);
        }
    } else {
        // 6.如果innerDone方法为false，即内层循环没有结束，则直接返回内层迭代器的next方法返回值。
        return innerIter.next();
    }
}

这个方法涉及到了内层迭代。首先判断内层迭代有没有结束。第一次进入这个方法时，连内层迭代器都还没有判断需不需要创建，其值为null，所以返回的一定是true。

private boolean innerDone() {
    return innerIter == null || !innerIter.hasNext();
}

接下来进入到if为true的分支。可以设想，这个分支里会判断是否需要给内层迭代器做初始化，是否需要内层迭代，如果需要的话则会开始内层迭代。这样的话，外层下一次调用hasNext方法进而进入到makeNext方法时，第一步innerDone的返回值就应该是false了，于是会来到第六步，直接返回内层迭代器的next方法返回值。这只是从逻辑上分析了整个流程，下面进行更加细致的分析。

private LogEntry getNextEntryFromStream() throws IOException {
    // 1.读取8个字节，获取offset。
    long offset = stream.readLong();
    // 2.读取4个字节，获取size。
    int size = stream.readInt();
  
  ......
    
    // 3.获取message data部分。
    ByteBuffer rec;
    if (type == CompressionType.NONE) {
        rec = buffer.slice();
        int newPos = buffer.position() + size;
        if (newPos > buffer.limit())
            return null;
        buffer.position(newPos);
        rec.limit(size);
    } else {
        byte[] recordBuffer = new byte[size];
        stream.readFully(recordBuffer, 0, size);
        rec = ByteBuffer.wrap(recordBuffer);
    }
    // 4.封装成LogEntry对象返回。
    return new LogEntry(offset, new Record(rec));
}

读取数据的过程相当于就是写入的逆过程。这里的offset是绝对的offset，而不是生产者生产时设置的相对offset，由此我们知道消息的offset值在Server端进行了重新分配，所以Consumer端可以直接读取到真实的offset值。这里会根据message data部分有没有经过压缩选择不同的读取方式。如果未经过压缩，可以直接根据size大小把buffer中相应的数据取出来；如果经过了压缩，则需要用stream对象的read相关方法去读取，这个stream对象封装了压缩类型信息。因为外层的这部分数据没有经过压缩（压缩的是内层的message data），所以外层的type是NONE。最后封装一个LogEntry对象返回。下图帮助回顾消息的格式：

public T next() {

        throw new NoSuchElementException();

这样就完整地完成了一次for循环。注意到next方法也调用了hasNext方法，同时也对state状态做了修改。这里我们不需要将注意力分散到这些操作，正常情况下都是直接返回next字段。

private RecordsIterator(LogEntry entry) {
    this.type = entry.record().compressionType();
    this.buffer = entry.record().value();
    this.shallow = true;
    this.stream = Compressor.wrapForInput(new ByteBufferInputStream(this.buffer), type, entry.record().magic());
    long wrapperRecordOffset = entry.offset();
    // If relative offset is used, we need to decompress the entire message first to compute
    // the absolute offset.
    if (entry.record().magic() > Record.MAGIC_VALUE_V0) {
        this.logEntries = new ArrayDeque<>();
        long wrapperRecordTimestamp = entry.record().timestamp();
        while (true) {
            try {
                LogEntry logEntry = getNextEntryFromStream();
                Record recordWithTimestamp = new Record(logEntry.record().buffer(),
                                                        wrapperRecordTimestamp,
                                                        entry.record().timestampType());
                logEntries.add(new LogEntry(logEntry.offset(), recordWithTimestamp));
            } catch (EOFException e) {
                break;
            } catch (IOException e) {
                throw new KafkaException(e);
            }
        }
        this.absoluteBaseOffset = wrapperRecordOffset - logEntries.getLast().offset();
    } else {
        this.logEntries = null;
        this.absoluteBaseOffset = -1;
    }

}

这个内层的迭代器的多个字段都与外层的值不一样。type不再是NONE，而是内层消息的压缩类型。shallow为true，表示这个迭代器已经是内层迭代器了，它不会再进行更深层的迭代。stream对象的构造方式值得注意：

public static DataInputStream wrapForInput(ByteBufferInputStream buffer, CompressionType type, byte messageVersion) {
    try {
        switch (type) {
            case NONE:
                return new DataInputStream(buffer);
            case GZIP:
                return new DataInputStream(new GZIPInputStream(buffer));
            case SNAPPY:
                try {
                    InputStream stream = (InputStream) snappyInputStreamSupplier.get().newInstance(buffer);
                    return new DataInputStream(stream);
                } catch (Exception e) {
                    throw new KafkaException(e);
                }
            case LZ4:
                try {
                    InputStream stream = (InputStream) lz4InputStreamSupplier.get().newInstance(buffer,
                            messageVersion == Record.MAGIC_VALUE_V0);
                    return new DataInputStream(stream);
                } catch (Exception e) {
                    throw new KafkaException(e);
                }
            default:
                throw new IllegalArgumentException("Unknown compression type: " + type);
        }
    } catch (IOException e) {
        throw new KafkaException(e);
    }
}

protected LogEntry makeNext() {
    if (innerDone()) {
        try {
            LogEntry entry = getNextEntry();
            if (entry == null)
                return allDone();

            if (absoluteBaseOffset >= 0) {
                long absoluteOffset = absoluteBaseOffset + entry.offset();
                entry = new LogEntry(absoluteOffset, entry.record());
            }

            CompressionType compression = entry.record().compressionType();
            if (compression == CompressionType.NONE || shallow) {
                return entry;
            } else {
                innerIter = new RecordsIterator(entry);
                return innerIter.next();
            }
        } catch (EOFException e) {
            return allDone();
        } catch (IOException e) {
            throw new KafkaException(e);
        }
    } else {
        return innerIter.next();
    }
}

public boolean hasNext() {
    switch (state) {
        case FAILED:
            throw new IllegalStateException("Iterator is in failed state");
        case DONE:
            return false;
        case READY:
            return true;
        default:
            return maybeComputeNext();
    }
}

当给next赋值后，会将状态设置为READY，所以再次进入hasNext方法时，就会直接返回true。next方法中会将state设置为NOT_READY，所以下一次调用hasNext方法时又会进入到maybeComputeNext方法去取下一个元素给next字段赋值。这就是为什么要有state状态。
